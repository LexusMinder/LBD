SELECT Profesor.Titulo, Cuenta_profesor.Profesor_Empleado_No_empleado
FROM Profesor
LEFT JOIN Cuenta_profesor ON Profesor.Empleado_No_empleado = Cuenta_profesor.Profesor_Empleado_No_empleado
ORDER BY Profesor.Titulo
